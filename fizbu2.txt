1. Architecture Overview
Fizbo runs as a standalone Next.js 15 application within the FAIRBANK Group monorepo. It has its own deployment on Vercel, its own domain (fizbo.pt), and its own set of Supabase tables — but shares the group's authentication system, Supabase instance, and property record schema.
 
Key constraint:  Fizbo's team does not own or modify PropCheck's valuation engine or Reserva's matching engine. These are consumed as internal API dependencies. Fizbo calls them; it does not build them.

 
FIZBO APPLICATION BOUNDARIES

Fizbo OWNS:
  apps/fizbo/                  — Next.js 15 app (App Router)
  services/api/doccheck/       — DocCheck decision tree (FastAPI)
  services/api/ocr/            — Document OCR pipeline (FastAPI + Mistral)
  services/api/workflows/      — Inngest workflow definitions
  supabase/migrations/fizbo/   — Fizbo-specific table migrations

Fizbo CONSUMES (do not modify):
  packages/auth/               — Shared Supabase auth
  packages/ui/                 — Shared component library
  packages/valuation/          — PropCheck AVM (internal API call)
  packages/notifications/      — Shared Resend + 360dialog
  property_intelligence schema — Shared property record (read/write)
  personal_data schema         — Shared user records (read/write)

5. Inngest Workflow Definitions
5.1  DocReady Standard Workflow
export const docreadyStandard = inngest.createFunction(
  { id: "docready-standard", retries: 3 },
  { event: "docready/order.created" },
  async ({ event, step }) => {
    const { orderId, tier, sellerId, propertyId } = event.data

    // Step 1: Generate document checklist (automated — DocCheck engine)
    const requirements = await step.run("generate-checklist", async () => {
      const result = await doccheckEngine(propertyId)
      await supabase.from("fizbo_documents").insert(
        result.documents.map(d => ({ order_id: orderId, property_id: propertyId, document_type: d.type, status: "missing" }))
      )
      return result
    })

    // Step 2: Notify operator (human-in-the-loop begins)
    await step.run("notify-operator", async () => {
      await supabase.from("platform_ops.operator_queue").insert({
        order_id: orderId, task_type: "process_order", status: "pending", priority: tierToPriority(tier)
      })
      await notifyOperator({ orderId, requirements })
    })

    // Step 3: Wait for each document (timeout: 30 days per document)
    for (const doc of requirements.documents) {
      await step.waitForEvent(`doc-complete-${orderId}-${doc.type}`, {
        event: "document/status.updated",
        if: `async.data.documentId == "${doc.id}" && async.data.status == "complete"`,
        timeout: "30d",
      })
    }

    // Step 4: Generate SmartCMA and complete order
    await step.run("complete-order", async () => {
      await generateSmartCMA(propertyId, sellerId)
      await supabase.from("fizbo_orders")
        .update({ status: "complete", completed_at: new Date() })
        .eq("id", orderId)
      await notifySeller(sellerId, "ORDER_COMPLETE")
    })
  }
)

// Operator triggers document completion:
export async function markDocumentComplete(documentId: string) {
  await supabase.from("fizbo_documents")
    .update({ status: "complete", obtained_at: new Date() })
    .eq("id", documentId)
  // This fires the Inngest waitForEvent above
  await inngest.send({
    name: "document/status.updated",
    data: { documentId, status: "complete" }
  })
  await notifySeller(sellerId, "DOCUMENT_COMPLETE")
}

// Runs daily at 08:00 Lisbon time
export const docExpiryCron = inngest.createFunction(
  { id: "doc-expiry-check" },
  { cron: "0 8 * * *" },
  async ({ step }) => {
    // Check for documents expiring in 30 days
    const expiring30 = await step.run("check-30-day", async () => {
      return supabase.from("fizbo_documents")
        .select("*, fizbo_orders(seller_id)")
        .lte("expiry_date", addDays(new Date(), 30))
        .gte("expiry_date", addDays(new Date(), 28))
        .eq("status", "complete")
    })
    for (const doc of expiring30.data) {
      await notifySeller(doc.fizbo_orders.seller_id, "DOCUMENT_EXPIRING_30_DAYS", { docType: doc.document_type })
    }
    // Same pattern for 7-day expiry (DOCUMENT_EXPIRING_7_DAYS)
  }
)

 

6. DocCheck Engine (Python)
# services/api/doccheck/engine.py

from dataclasses import dataclass
from enum import Enum

class DocumentType(str, Enum):
    CADERNETA = "caderneta_predial"
    CERTIDAO = "certidao_permanente"
    LICENCA = "licenca_utilizacao"
    FICHA_TECNICA = "ficha_tecnica"
    CERT_ENERGETICO = "cert_energetico"
    DECLARACAO_CONDO = "declaracao_condominio"
    DISTRATE = "distrate_hipoteca"
    HABILITACAO = "habilitacao_herdeiros"

@dataclass
class DocCheckAnswers:
    property_type: str           # apartamento|moradia|terreno|outro
    has_condominium: str         # yes|no|unsure
    construction_era: str        # pre1951|1951_1990|1991_2007|post2007
    has_mortgage: bool
    acquisition_type: str        # purchase|inheritance|divorce|gift|other
    is_primary_residence: str    # yes|no_rented|no_vacant
    has_energy_cert: str         # yes|no|unknown
    all_owners_available: str    # yes|one_abroad|one_deceased|disputed

def evaluate(answers: DocCheckAnswers) -> DocCheckResult:
    docs = []

    # Always required
    docs.append({"type": DocumentType.CADERNETA, "required": True})
    docs.append({"type": DocumentType.CERTIDAO, "required": True})

    # Conditional on construction era
    if answers.construction_era != "pre1951":
        docs.append({"type": DocumentType.LICENCA, "required": True})
    if answers.construction_era == "post2007":
        docs.append({"type": DocumentType.FICHA_TECNICA, "required": True})

    # Energy certificate
    if answers.has_energy_cert != "yes":
        docs.append({"type": DocumentType.CERT_ENERGETICO, "required": True})

    # Condominium declaration
    if answers.has_condominium == "yes":
        docs.append({"type": DocumentType.DECLARACAO_CONDO, "required": True})

    # Mortgage discharge
    if answers.has_mortgage:
        docs.append({"type": DocumentType.DISTRATE, "required": True})

    # Inheritance
    if answers.acquisition_type == "inheritance" and answers.all_owners_available != "yes":
        docs.append({"type": DocumentType.HABILITACAO, "required": True, "is_prerequisite": True})

    # Determine tier
    required_count = sum(1 for d in docs if d["required"])
    has_prerequisite = any(d.get("is_prerequisite") for d in docs)

    if required_count <= 2 and not has_prerequisite:
        tier = "standard"
    elif required_count >= 3 or has_prerequisite:
        tier = "premium"
    if answers.all_owners_available == "disputed" or answers.acquisition_type == "divorce":
        tier = "express"

    risk_flags = build_risk_flags(answers)

    return DocCheckResult(documents=docs, recommended_tier=tier, risk_flags=risk_flags)

 

7. OCR Pipeline
7.1  Caderneta Predial OCR Flow
# services/api/ocr/caderneta.py

async def process_caderneta(file_bytes: bytes, content_type: str) -> CadernetaResult:
    # Step 1: Call Mistral OCR
    import base64
    b64 = base64.b64encode(file_bytes).decode()
    response = await mistral_client.chat.complete(
        model="mistral-ocr-latest",
        messages=[{
            "role": "user",
            "content": [
                {"type": "image_url", "image_url": {"url": f"data:{content_type};base64,{b64}"}},
                {"type": "text", "text": CADERNETA_EXTRACTION_PROMPT}
            ]
        }]
    )

    # Step 2: Parse structured fields
    raw_text = response.choices[0].message.content
    extracted = parse_caderneta_fields(raw_text)

    # Step 3: Validate critical fields
    if not extracted.artigo_matricial or not extracted.municipality:
        raise OCRExtractionError("Critical fields missing")

    return extracted

CADERNETA_EXTRACTION_PROMPT = """
Extract the following fields from this Portuguese Caderneta Predial Urbana.
Return ONLY a JSON object with these exact keys:
{"artigo_matricial": "", "address": "", "parish": "", "municipality": "",
 "typology": "", "gross_area": 0, "vpt": 0, "owner_name": ""}
If a field is not visible, use null. Do not include any other text.
"""

// apps/fizbo/middleware.ts

export async function middleware(request: NextRequest) {
  const { session, profile } = await getSession()

  // Public routes — no auth required
  const publicPaths = ["/", "/verificar", "/encomendar", "/precos", "/en/*"]
  if (publicPaths.some(p => request.nextUrl.pathname.startsWith(p))) return

  // Auth required for /dashboard/*
  if (!session) return NextResponse.redirect(new URL("/login", request.url))

  // Admin required for /admin/*
  if (request.nextUrl.pathname.startsWith("/admin")) {
    if (profile?.user_type !== "admin") {
      return NextResponse.redirect(new URL("/dashboard", request.url))
    }
  }
}

8.1  GDPR Implementation
NIF stored encrypted: INSERT INTO personal_data.user_profiles ... WITH nif = pgp_sym_encrypt($1, $2)
Document files: Supabase Storage private bucket "fizbo-documents". All access via signed URLs (1hr expiry).
Right-to-deletion: DELETE /api/gdpr/delete — removes all personal_data records, Storage files, and sets order records to anonymised state.
Data export: GET /api/gdpr/export — returns ZIP of all personal data for a user.
